        this.cantidad = cantidad;

    let Factura = function(empresa,direccion,telefono,nif,f_pago,elementos){
        this.empresa = empresa;
        this.direccion = direccion;
        this.telefono = telefono;
        this.nif = nif;
        this.total = 0;
        this.t_iva = 0;
        this.f_pago = f_pago;
        this.elementos = elementos;
        this.calcularTotal = function(){
            var fin = this.elementos.length;
            for(let i = 0; i < fin; i++){
                this.total += parseFloat(this.elementos[i]['precio']) * parseFloat(this.elementos[i]['cantidad']);
            }
            this.calcularIva();
        }
        this.calcularIva = function(){
            this.t_iva = ( this.total / 100 ) * 21;
        }
    }

let DetectarIdioma = function (){
        this.idiomas = [
            { idioma: "Español",texto: "Bienvenido al curso de React Native en PachaQtec",id_idioma: 1,},
            { idioma: "Ingles",texto: "Welcome to the React Native course at PachaQtec",id_idioma: 2,},
            { idioma: "Francés",texto: "Bienvenue dans le cours React Native chez PachaQtec",id_idioma: 3,},
            { idioma: "Italiano",texto: "Benvenuti al corso React Native presso PachaQtec",id_idioma: 4,},
            { idioma: "Portugués",texto: "Bem-vindo ao curso React Native na PachaQtec",id_idioma: 5,}
        ];
        this.texto = '';
        this.idioma = '';
        this.id_idioma = 0;
        this.posiblesIdiomas = new Array();
        this.detectarIdioma = function(texto){
            this.texto = texto;
            let fin = this.idiomas.length;
            for(let i = 0; i < fin; i++){
                if( this.idiomas[i]['texto'].indexOf(this.texto) != -1 ){
                    this.posiblesIdiomas.push( {idioma:this.idiomas[i]['idioma'],id_idioma:this.idiomas[i]['id_idioma']} );
                }
            }
        }
    }

    detectar = new DetectarIdioma();
    detectar.detectarIdioma('Bienvenido');
    document.getElementById("result8").innerHTML = 'ver consola';
    console.log(detectar.posiblesIdiomas);
